title Smart Elevator Control
buildings [icon: home, color: blue] {
  building_id serial pk
  name varchar(200)
  address text
  total_floors int
}
floors [icon: list, color: green] {
  floor_id serial pk
  building_id int fk
  floor_number int
  floor_name varchar(50)
}

elevator_shafts [icon: server, color: gray] {
  shaft_id serial pk
  building_id int fk
  shaft_name varchar(50)
}

elevators [icon: arrow-up, color: orange] {
  elevator_id serial pk
  shaft_id int fk
  current_floor_id int fk
  status enum('idle', 'moving_up', 'moving_down', 'maintenance')
  capacity_kg int
}

elevator_floor_access [icon: lock, color: yellow] {
  access_id serial pk
  elevator_id int fk
  floor_id int fk
}

floor_requests [icon: bell, color: red] {
  request_id serial pk
  source_floor_id int fk
  direction enum('up', 'down')
  request_time timestamp
  status enum('pending', 'assigned', 'completed')
}

ride_assignments [icon: activity, color: purple] {
  ride_id serial pk
  request_id int fk
  elevator_id int fk
  pickup_time timestamp
  dropoff_time timestamp
  ride_status enum('in_progress', 'completed', 'failed')
}

maintenance_logs [icon: settings, color: red] {
  log_id serial pk
  elevator_id int fk
  issue_description text
  start_time timestamp
  end_time timestamp
  resolved_by varchar(100)
}

// Ek building mein kai floors aur shafts hote hain
buildings.building_id < floors.building_id:[color: yellow]
buildings.building_id < elevator_shafts.building_id:[color: yellow]

// Ek shaft mein ek hi elevator fit hota hai
elevator_shafts.shaft_id - elevators.shaft_id:[color: yellow]

// Many-to-Many: Ek elevator kin-kin floors par ruk sakta hai
elevators.elevator_id < elevator_floor_access.elevator_id:[color: yellow]
floors.floor_id < elevator_floor_access.floor_id:[color: yellow]

// Ek floor se kai requests aati hain din bhar mein
floors.floor_id < floor_requests.source_floor_id:[color: yellow]

// Ek request ek elevator ko assign ki jati hai ride ke roop mein
floor_requests.request_id - ride_assignments.request_id
elevators.elevator_id < ride_assignments.elevator_id:[color: yellow]

// Ek elevator ki multiple maintenance history hoti hai
elevators.elevator_id < maintenance_logs.elevator_id:[color: yellow]
